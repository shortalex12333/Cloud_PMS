## Executive Summary

This document specifies the complete frontend customer journey for the handover export system, covering:
1. Email extraction from Outlook
2. Handover draft review and acceptance
3. Dual sign-off workflow
4. Export and distribution

All designs adhere to CelesteOS UX principles: **search-first interface**, **no dashboards**, **explicit user confirmation**.

## Part 2: Search Integration

### Search Intent Patterns

Following CelesteOS's **search-as-interface** principle:

| User Input | Detected Intent | Action |
|------------|-----------------|--------|
| "extract emails" | email_extraction | Open email extraction flow |
| "handover emails" | email_extraction | Open email extraction flow |
| "generate handover" | handover_generation | Start draft generation |
| "show handover" | handover_view | Show current/pending handovers |
| "sign handover" | handover_sign | Go to sign-off screen |
| "countersign" | handover_countersign | Go to countersign screen |
| "export handover" | handover_export | Open export options |
| "handover history" | handover_history | Show past signed handovers |

### Search Results for Handover

When user searches "handover", return action cards:

```typescript
interface HandoverSearchResult {
  type: 'action_card';
  actions: [
```

```typescript
// Email extraction
const useEmailExtraction = () => {
  const startExtraction = useMutation({
    mutationFn: (config: ExtractionConfig) =>
      api.post('/api/v1/email-handover/extract', config),
    onSuccess: (data) => {
      // Start polling for job status
      setJobId(data.job_id);
    }
  });

  const jobStatus = useQuery({
    queryKey: ['extraction-job', jobId],
    queryFn: () => api.get(`/api/v1/email-handover/jobs/${jobId}`),
    refetchInterval: jobId ? 2000 : false,
    enabled: !!jobId
  });

  return { startExtraction, jobStatus };
};

// Handover draft
const useHandoverDraft = (draftId: string) => {
  const draft = useQuery({
    queryKey: ['handover-draft', draftId],
    queryFn: () => api.get(`/handover/draft/${draftId}`)
  });

  const submitForReview = useMutation({
    mutationFn: () => api.post(`/handover/draft/${draftId}/review`)
  });

  const acceptHandover = useMutation({
    mutationFn: (acknowledgment: string) =>
      api.post(`/handover/draft/${draftId}/accept`, { acknowledgment })
  });

  const countersignHandover = useMutation({
    mutationFn: (acknowledgment: string) =>
      api.post(`/handover/draft/${draftId}/sign`, { acknowledgment })
  });

  return { draft, submitForReview, acceptHandover, countersignHandover };
};

// Export
const useHandoverExport = (draftId: string) => {
  const exportPDF = useMutation({
    mutationFn: () =>
      api.post(`/handover/draft/${draftId}/export`, { export_type: 'pdf' })
  });

  const sendEmail = useMutation({
    mutationFn: (recipients: string[]) =>
      api.post(`/handover/draft/${draftId}/export`, {
        export_type: 'email',
        recipients
      })
  });

  return { exportPDF, sendEmail };
};
```

## Part 8: State Machine (Frontend)

```typescript
// Handover state machine for frontend

type HandoverState =
  | 'DRAFT'
  | 'IN_REVIEW'
  | 'ACCEPTED'
  | 'SIGNED'
  | 'EXPORTED';

type HandoverEvent =
  | { type: 'SUBMIT_FOR_REVIEW' }
  | { type: 'ACCEPT'; payload: { acknowledgment: string } }
  | { type: 'COUNTERSIGN'; payload: { acknowledgment: string } }
  | { type: 'EXPORT'; payload: { format: 'pdf' | 'html' | 'email' } }
  | { type: 'ADD_ITEM' }
  | { type: 'EDIT_ITEM'; payload: { itemId: string } };

const handoverMachine = createMachine<HandoverState, HandoverEvent>({
  initial: 'DRAFT',
  states: {
    DRAFT: {
      on: {
        SUBMIT_FOR_REVIEW: 'IN_REVIEW',
        ADD_ITEM: 'DRAFT',
        EDIT_ITEM: 'DRAFT'
      }
    },
    IN_REVIEW: {
      on: {
        ACCEPT: {
          target: 'ACCEPTED',
          actions: ['signOutgoing']
        },
        ADD_ITEM: 'IN_REVIEW'
        // No EDIT_ITEM - locked
      }
    },
    ACCEPTED: {
      on: {
        COUNTERSIGN: {
          target: 'SIGNED',
          actions: ['signIncoming']
        }
        // No modifications allowed
      }
    },
    SIGNED: {
      on: {
        EXPORT: {
          target: 'EXPORTED',
          actions: ['generateExport']
        }
      }
    },
    EXPORTED: {
      type: 'final'
    }
  }
});
```
